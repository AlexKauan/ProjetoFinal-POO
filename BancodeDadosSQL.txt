create table vendedor(
	id_vendedor serial,
	login varchar(50),
	senha varchar(50),
	nome varchar(50),
	telefone varchar(50),
	email varchar(50),
	endereco varchar(50),
	salario numeric(50,2),
	numeroDeVendas int,
	totalVendido numeric(50,2),
	primary key (id_vendedor)
);

select * from vendedor

create table cliente(
	id_cliente serial,
	login varchar(50),
	senha varchar(50),
	nome varchar(50),
	telefone varchar(50),
	email varchar(50),
	endereco varchar(50),
	numero_de_compras int,
	totalcomprado numeric(50,2),
	primary key (id_cliente)
);

create table compra(
	id_compra serial,
	id_cliente int,
	data varchar(50),
	statusCompra varchar(50),
	primary key (id_compra),
	foreign key (id_cliente) references cliente (id_cliente)
);
